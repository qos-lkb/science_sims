    async function onConfigChange(config) {
      const mainTitle = document.getElementById('main-title');
      const subtitle = document.getElementById('subtitle');
      const registrationButton = document.getElementById('registration-button');
      const contactTitle = document.getElementById('contact-title');
      
      if (mainTitle) mainTitle.textContent = config.main_title || defaultConfig.main_title;
      if (subtitle) subtitle.textContent = config.subtitle || defaultConfig.subtitle;
      if (registrationButton) registrationButton.textContent = config.registration_button || defaultConfig.registration_button;
      if (contactTitle) contactTitle.textContent = config.contact_title || defaultConfig.contact_title;
      
      const primaryColor = config.primary_color || defaultConfig.primary_color;
      const secondaryColor = config.secondary_color || defaultConfig.secondary_color;
      const accentColor = config.accent_color || defaultConfig.accent_color;
      const backgroundColor = config.background_color || defaultConfig.background_color;
      const textColor = config.text_color || defaultConfig.text_color;
      const fontFamily = config.font_family || defaultConfig.font_family;
      const fontSize = config.font_size || defaultConfig.font_size;
      
      // Update hero gradient
      const heroSection = document.querySelector('.hero-section');
      if (heroSection) {
        heroSection.style.background = `linear-gradient(135deg, ${primaryColor} 0%, ${accentColor} 50%, ${secondaryColor} 100%)`;
      }
      
      // Update section titles
      const sectionTitles = document.querySelectorAll('.section-title, .criteria-label, .criteria-value, .info-box-title, .requirement-title, .timeline-date, .award-name');
      sectionTitles.forEach(el => {
        el.style.color = primaryColor;
      });
      
      // Update CTA button
      const ctaButton = document.querySelector('.cta-button');
      if (ctaButton) {
        ctaButton.style.background = `linear-gradient(135deg, ${primaryColor} 0%, ${secondaryColor} 100%)`;
      }
      
      // Update progress bars with distinct colors
      const segment1 = document.getElementById('segment-1');
      const segment2 = document.getElementById('segment-2');
      const segment3 = document.getElementById('segment-3');
      
      if (segment1) segment1.style.background = `linear-gradient(135deg, ${primaryColor} 0%, ${accentColor} 100%)`;
      if (segment2) segment2.style.background = `linear-gradient(135deg, #ff6b6b 0%, #ee5a6f 100%)`;
      if (segment3) segment3.style.background = `linear-gradient(135deg, #4ecdc4 0%, #44a8a0 100%)`;

      
      // Update timeline
      const timeline = document.querySelector('.timeline::before');
      const timelineItems = document.querySelectorAll('.timeline-item::before');
      document.querySelectorAll('.timeline-item').forEach(item => {
        const style = document.createElement('style');
        style.textContent = `
          .timeline::before { background: linear-gradient(to bottom, ${primaryColor}, ${secondaryColor}) !important; }
          .timeline-item::before { background: ${primaryColor} !important; box-shadow: 0 0 0 3px ${primaryColor} !important; }
        `;
        if (!document.getElementById('timeline-style')) {
          style.id = 'timeline-style';
          document.head.appendChild(style);
        } else {
          document.getElementById('timeline-style').textContent = style.textContent;
        }
      });
      
      // Update info boxes
      const infoBoxes = document.querySelectorAll('.info-box');
      infoBoxes.forEach(box => {
        box.style.borderLeftColor = primaryColor;
      });
      
      // Update requirement items
      const requirementItems = document.querySelectorAll('.requirement-item');
      requirementItems.forEach(item => {
        item.style.borderLeftColor = primaryColor;
      });
      
      // Update contact section
      const contactSection = document.querySelector('.contact-section');
      if (contactSection) {
        contactSection.style.background = `linear-gradient(135deg, ${primaryColor} 0%, ${accentColor} 100%)`;
      }
      
      // Update fonts
      document.body.style.fontFamily = `'${fontFamily}', 'Noto Sans TC', sans-serif`;
      document.body.style.fontSize = `${fontSize}px`;
      
      // Update heading sizes proportionally
      const heroTitle = document.querySelector('.hero-title');
      if (heroTitle) heroTitle.style.fontSize = `${fontSize * 3.5}px`;
      
      const heroSubtitle = document.querySelector('.hero-subtitle');
      if (heroSubtitle) heroSubtitle.style.fontSize = `${fontSize * 1.75}px`;
      
      const heroDescription = document.querySelector('.hero-description');
      if (heroDescription) heroDescription.style.fontSize = `${fontSize * 1.125}px`;
      
      const mainSectionTitles = document.querySelectorAll('.section-title');
      mainSectionTitles.forEach(title => {
        title.style.fontSize = `${fontSize * 2.25}px`;
      });
    }

    function mapToCapabilities(config) {
      return {
        recolorables: [
          {
            get: () => config.primary_color || defaultConfig.primary_color,
            set: (value) => {
              config.primary_color = value;
              window.elementSdk.setConfig({ primary_color: value });
            }
          },
          {
            get: () => config.secondary_color || defaultConfig.secondary_color,
            set: (value) => {
              config.secondary_color = value;
              window.elementSdk.setConfig({ secondary_color: value });
            }
          },
          {
            get: () => config.accent_color || defaultConfig.accent_color,
            set: (value) => {
              config.accent_color = value;
              window.elementSdk.setConfig({ accent_color: value });
            }
          },
          {
            get: () => config.background_color || defaultConfig.background_color,
            set: (value) => {
              config.background_color = value;
              window.elementSdk.setConfig({ background_color: value });
            }
          },
          {
            get: () => config.text_color || defaultConfig.text_color,
            set: (value) => {
              config.text_color = value;
              window.elementSdk.setConfig({ text_color: value });
            }
          }
        ],
        borderables: [],
        fontEditable: {
          get: () => config.font_family || defaultConfig.font_family,
          set: (value) => {
            config.font_family = value;
            window.elementSdk.setConfig({ font_family: value });
          }
        },
        fontSizeable: {
          get: () => config.font_size || defaultConfig.font_size,
          set: (value) => {
            config.font_size = value;
            window.elementSdk.setConfig({ font_size: value });
          }
        }
      };
    }

    function mapToEditPanelValues(config) {

        ["subtitle", config.subtitle || defaultConfig.subtitle],

        ["contact_title", config.contact_title || defaultConfig.contact_title]

    }

    if (window.elementSdk) {
      window.elementSdk.init({

        mapToCapabilities,
        mapToEditPanelValues
